int max(int a, int b) {
  return (a > b) ? a : b;
}

int longest_com_seq(int *s1, int *s2, int n, int m) {
  int x[n+1][m+1];
  for (int i = n; i >= 0; i--) {
    for (int j = m; j >= 0; j--) {
      if (i == n || j == m) x[i][j] = 0;
      else if (s1[i] == s2[j]) x[i][j] = x[i+1][j+1] + 1;
      else x[i][j] = max(x[i][j+1], x[i+1][j]);
    }
  }
  return x[0][0];
}

#define ll long long

ll longest_inc_seq(int *a, int n) {
  ll ans, m[n];

  for(int i = n-1; i >= 0; i--) {
    m[i] = 1;
    for(int j = i+1; j < n; j++) 
      if(a[j] > a[i] && m[i] < m[j] + 1)
          m[i] = m[j] + 1;
  }

  for(int i = 0; i < n; i++) 
    if (ans < m[i])
      ans = m[i];

  return ans;
}

int *subset_sum(int *a, int n) {
  int M = 0;
  for (int i = 0; i < n; i++) M += a[i];

  int *m = (int*) malloc(sizeof(int) * (M+1)); 
  for (int i = 0; i <= M; i++) m[i] = 0;

  m[0] = 1;
  for (int i = 0; i < n; i++) 
    for (int j = M; j >= a[i]; j--) 
      m[j] |= m[j-a[i]];
  return m;
}

int subset_sum_k(int *a, int n, int k) {
  int M = 0;
  for (int i = 0; i < n; i++) M += a[i];

  int m[M+1]; 
  for (int i = 0; i <= M; i++) m[i] = 0;

  m[0] = 1;
  for (int i = 0; i < n; i++) 
    for (int j = M; j >= a[i]; j--) 
      m[j] |= m[j-a[i]];

  return m[k];
}

#include <stdlib.h>

using namespace std;

#define ll long long

ll max(ll a, ll b) {
  return (a > b) ? a : b;
}

int bin_search(int i, int start, int end, int *finishTimeIndex, int *finishTime, int *startTime) {
  if (start > end)
    return end;
  int mid = start + ((end-start)/2);
  if (finishTime[finishTimeIndex[mid]] < startTime[finishTimeIndex[i]])
    return bin_search(i, mid+1, end, finishTimeIndex, finishTime, startTime);
  else
    return bin_search(i, start, mid-1, finishTimeIndex, finishTime, startTime);
  return mid;
}

ll WeightedActivitySelection(int n, int *w, int *finishTimeIndex, int *finishTime, int *startTime) {
  ll opt[n];
  opt[0] = w[finishTimeIndex[0]];
  for (int i = 1; i < n; i++) { 
    ll t = bin_search(i, 0, n-1, finishTimeIndex, finishTime, startTime); 
    if (t != -1)
      opt[i] = max(opt[i-1], opt[t] + w[finishTimeIndex[i]]);
    else
      opt[i] = max(opt[i-1], w[finishTimeIndex[i]]);
  }
  return opt[n-1];
}
